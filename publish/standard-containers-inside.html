class Vector<bool> {
private:
    size_t sz;
    size_t cap;
    int8_t* arr;
public:
    Vector(size_t sz, bool default_value): sz(sz), cap(sz / 8 + ((sz % 8 != 0) ? 1 : 0)), arr(new int8_t[sz / 8 + ((sz % 8 != 0) ? 1 : 0)]) {
        if (default_value) {
            for (size_t i = 0; i < cap; ++i) {
                arr[i] |= 11111111u;
            }
        } else {
            for (size_t i = 0; i < cap; ++i) {
                arr[i] &= 0u;
            }
        }
    }

    ~Vector() {
        delete[] arr;
    }

    struct BitReference {
        int8_t* cell;
        uint8_t pos;

        BitReference(int8_t* cell, uint8_t pos): cell(cell), pos(pos) {}

        operator bool() {
            return *cell & (1u << pos);
        }

        BitReference& operator=(bool value) {
            if (value) {
                *cell |= (1u << pos);
            } else {
                *cell &= ~(1u << pos);
            }
            return *this;
        }
    };

    BitReference operator[](size_t index) {
        return BitReference(arr + index / 8, index % 8);
    }

    const BitReference operator[](size_t index) const {
        return BitReference(arr + index / 8, index % 8);
    }
};
