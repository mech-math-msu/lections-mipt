Управляющие конструкции

Условный оператор (if else)

#include <iostream>

int main()
{
    // if (bool-expression) {// expression - это какое-то выражение,
    //     statements   // возвращающие булев тип, 
    // }                // если тип не тот, то компилятор
                        // попытается превратить его в bool)
    if (true) {
        std::cout << "in if\n";
    } else {
        std::cout << "i think i can't be here\n";
    }
    // так как true всегда true, в else программа никогда не попадет
}

Switch Case

#include <iostream>

int main()
{
    switch (expression) {
        case 1:
            statements; // инструкции через точку с запятой
            break; // выход из switch, если не написать, то будет выполнять все что дальше
        case 2:
            statements;
            break;
        default: // будет выполнен, если не выполнился ни один из кейсов
            statements;
    }
}

Циклы

#include <iostream>
#include <vector>

int main()
{
    // for (declaration or expression; bool-expression; expression) {
    //     statements;
    //     break;
    //     continue;
    // }

    // 1. цикл for сначала выполняет declaration or expression,
    // 2. потом проверяет на истинность bool-expression,
    // 3. если истинно, то выполняются statements, а затем expression и пункт 2.
    // 4. иначе завершается цикл

    // break - сразу завершает цикл, expression не выполняется
    // continue - заставляет for сразу выполнить expression,
    // игнорируя все statemens после continue

    for (int i = 0; i < 10; ++i) {
        std::cout << i << "\n";
    }

    for (std::vector<int> v; v.size() < 10; v.push_back(3)) {
        std::cout << v.back() << "\n";
    }

    // цикл while

    // while (condition) {
    //     statements;
    // }
    // 1. проверяется истинность condition
    // 2. если истинно, то выполняются statements и пункт 1.
    // 3. иначе цикл завершается

    // цикл do while

    // do {
    //      statements;
    // } while (condition)

    // тоже самое, что while, только условие проверяется после цикла
}
